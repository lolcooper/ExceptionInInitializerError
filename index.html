public class SevenDayTimer extends JavaPlugin {

public static int RelTiming = new main().getConfig().getInt("Sunday.Time");
public static Calendar cal = Calendar.getInstance();
@SuppressWarnings("deprecation")
public static int min = cal.getTime().getMinutes();

@SuppressWarnings("deprecation")
public static void SevenDayStartTimer()
{
    Bukkit.getServer().getScheduler().scheduleAsyncRepeatingTask(new main(), new Runnable() {

        public void run()
        {

            if(min == RelTiming)
            {
                System.out.println("Sonntag!");
                min++;
                BreakSevenDayTimer();
            }
        }
    }, 0L, 20L);
}

@SuppressWarnings("deprecation")
private static void BreakSevenDayTimer()
{
    Bukkit.getServer().getScheduler().scheduleAsyncDelayedTask(new main(), new Runnable() {

        public void run()
        {
            Calendar cal = Calendar.getInstance();
            min = cal.getTime().getMinutes();
        }
    }, 20*60L);
}
